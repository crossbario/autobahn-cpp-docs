115 template <typename T>
116 inline T wamp_call_result::kw_argument(const std::string& key) const
117 {
118  if (m_kw_arguments.type != msgpack::type::MAP) {
119  throw msgpack::type_error();
120  }
121  for (std::size_t i = 0; i < m_kw_arguments.via.map.size; ++i) {
122  const msgpack::object_kv& kv = m_kw_arguments.via.map.ptr[i];
123  if (kv.key.type == msgpack::type::STR && key.size() == kv.key.via.str.size
124  && key.compare(0, key.size(), kv.key.via.str.ptr, kv.key.via.str.size) == 0)
125  {
126  return kv.val.as<T>();
127  }
128  }
129  throw std::out_of_range(key + " keyword argument doesn't exist");
130 }

132 template <typename T>
133 inline T wamp_call_result::kw_argument(const char* key) const
134 {
135  if (m_kw_arguments.type != msgpack::type::MAP) {
136  throw msgpack::type_error();
137  }
138  std::size_t key_size = strlen(key);
139  for (std::size_t i = 0; i < m_kw_arguments.via.map.size; ++i) {
140  const msgpack::object_kv& kv = m_kw_arguments.via.map.ptr[i];
141  if (kv.key.type == msgpack::type::STR && key_size == kv.key.via.str.size
142  && memcmp(key, kv.key.via.str.ptr, key_size) == 0)
143  {
144  return kv.val.as<T>();
145  }
146  }
147  throw std::out_of_range(std::string(key) + " keyword argument doesn't exist");
148 }
