61 template <class Socket>
62 boost::future<void> wamp_rawsocket_transport<Socket>::connect()
63 {
64  m_connect = boost::promise<void>(); // reset the promise
65 
66  if (m_socket.is_open()) {
67  m_connect.set_exception(boost::copy_exception(network_error("network transport already connected")));
68  return m_connect.get_future();
69  }
70 
71  std::weak_ptr<wamp_rawsocket_transport<Socket>> weak_self = this->shared_from_this();
72  auto connect_handler = [=](const boost::system::error_code& error_code) {
73  auto shared_self = weak_self.lock();
74  if (!shared_self) {
75  return;
76  }
77 
78  if (error_code) {
79  m_socket.close(); // async_connect will leave it open
80  m_connect.set_exception(boost::copy_exception(
81  std::system_error(error_code.value(), std::system_category(), "connect")));
82  return;
83  }
84 
85  // FIXME: There should be a wamp_rawsocket_properties object or something
86  // similar that can be passed in to help deal with overiding these
87  // handshake parameters.
88  //
89  // Send the initial handshake packet informing the server which
90  // serialization format we wish to use, and our maximum message size.
91  m_handshake_buffer[0] = 0x7F; // magic byte
92  m_handshake_buffer[1] = 0xF2; // we are ready to receive messages up to 2**24 octets and encoded using MsgPack
93  m_handshake_buffer[2] = 0x00; // reserved
94  m_handshake_buffer[3] = 0x00; // reserved
95 
96  boost::asio::write(
97  m_socket,
98  boost::asio::buffer(m_handshake_buffer, sizeof(m_handshake_buffer)));
99 
100  auto handshake_reply = [=](
101  const boost::system::error_code& error,
102  std::size_t bytes_transferred) {
103  auto shared_self = weak_self.lock();
104  if (shared_self) {
105  handshake_reply_handler(error, bytes_transferred);
106  }
107  };
108 
109  try {
110  // Read the 4-byte handshake reply from the server
111  boost::asio::async_read(
112  m_socket,
113  boost::asio::buffer(m_handshake_buffer, sizeof(m_handshake_buffer)),
114  handshake_reply);
115  } catch (const std::exception& e) {
116  m_connect.set_exception(boost::copy_exception(e));
117  }
118  };
119 
120  m_socket.async_connect(m_remote_endpoint, connect_handler);
121 
122  return m_connect.get_future();
123 }

144 template <class Socket>
145 void wamp_rawsocket_transport<Socket>::send_message(wamp_message&& message)
146 {
147  auto buffer = std::make_shared<msgpack::sbuffer>();
148  msgpack::packer<msgpack::sbuffer> packer(*buffer);
149  packer.pack(message.fields());
150 
151  // Write the length prefix as the message header.
152  uint32_t length = htonl(buffer->size());
153  boost::asio::write(m_socket, boost::asio::buffer(&length, sizeof(length)));
154 
155  // Write actual serialized message.
156  boost::asio::write(m_socket, boost::asio::buffer(buffer->data(), buffer->size()));
157 
158  if (m_debug_enabled) {
159  std::cerr << "TX message (" << buffer->size() << " octets) ..." << std::endl;
160  std::cerr << "TX message: " << message << std::endl;
161  }
162 }

228 template <class Socket>
229 void wamp_rawsocket_transport<Socket>::handshake_reply_handler(
230  const boost::system::error_code& error_code,
231  std::size_t /* bytes_transferred */)
232 {
233  if (error_code) {
234  if (m_debug_enabled) {
235  std::cerr << "rawsocket handshake error: " << error_code << std::endl;
236  }
237 
238  m_connect.set_exception(boost::copy_exception(
239  std::system_error(error_code.value(), std::system_category(), "async_read")));
240  return;
241  }
242 
243  if (m_debug_enabled) {
244  std::cerr << "RawSocket handshake reply received" << std::endl;
245  }
246 
247  if (m_handshake_buffer[0] != 0x7F) {
248  m_connect.set_exception(boost::copy_exception(protocol_error("invalid handshake frame")));
249  return;
250  }
251 
252  // Indicates that the handshake reply is an error.
253  if ((m_handshake_buffer[1] & 0x0F) == 0x00) {
254  uint32_t error = m_handshake_buffer[1] & 0xF0;
255  if (m_debug_enabled) {
256  std::cerr << "rawsocket handshake error: " << std::hex << error << std::endl;
257  }
258 
259  std::stringstream error_string;
260  if (error == 0x00) {
261  error_string << "illegal error code (" << error << ")";
262  } else if (error == 0x10) {
263  error_string << "serializer unsupported";
264  } else if (error == 0x20) {
265  error_string << "maximum message length unacceptable";
266  } else if (error == 0x30) {
267  error_string << "use of reserved bits (unsupported feature)";
268  } else if (error == 0x40) {
269  error_string << "maximum connection count reached";
270  } else {
271  error_string << "unknown/reserved error code (" << error << ")";
272  }
273 
274  m_connect.set_exception(boost::copy_exception(protocol_error(error_string.str())));
275  return;
276  }
277 
278  uint32_t serializer_type = (m_handshake_buffer[1] & 0x0F);
279  if (serializer_type == 0x01) {
280  m_connect.set_exception(boost::copy_exception(protocol_error("json currently not supported")));
281  } else if (serializer_type == 0x02) {
282  if (m_debug_enabled) {
283  std::cerr << "connect successful: valid handshake" << std::endl;
284  }
285  m_connect.set_value();
286  receive_message();
287  } else {
288  std::stringstream error_string;
289  error_string << "rawsocket handshake error: invalid serializer type (" << serializer_type << ")";
290  m_connect.set_exception(boost::copy_exception(protocol_error(error_string.str())));
291  }
292 }

295 void wamp_rawsocket_transport<Socket>::receive_message()
296 {
297  if (m_debug_enabled) {
298  std::cerr << "RX preparing to receive message .." << std::endl;
299  }
300 
301  boost::asio::async_read(
302  m_socket,
303  boost::asio::buffer(&m_message_length, sizeof(m_message_length)),
304  bind(&wamp_rawsocket_transport<Socket>::receive_message_header,
305  this->shared_from_this(),
306  boost::asio::placeholders::error,
307  boost::asio::placeholders::bytes_transferred));
308 }

310 template <class Socket>
311 void wamp_rawsocket_transport<Socket>::receive_message_header(
312  const boost::system::error_code& error_code,
313  std::size_t /* bytes transferred */)
314 {
315  if (!error_code) {
316  m_message_length = ntohl(m_message_length);
317 
318  if (m_debug_enabled) {
319  std::cerr << "RX message (" << m_message_length << " octets) ..." << std::endl;
320  }
321 
322  m_message_unpacker.reserve_buffer(m_message_length);
323 
324  boost::asio::async_read(
325  m_socket,
326  boost::asio::buffer(m_message_unpacker.buffer(), m_message_length),
327  bind(&wamp_rawsocket_transport<Socket>::receive_message_body,
328  this->shared_from_this(),
329  boost::asio::placeholders::error,
330  boost::asio::placeholders::bytes_transferred));
331  return;
332  }
333 }

335 template <class Socket>
336 void wamp_rawsocket_transport<Socket>::receive_message_body(
337  const boost::system::error_code& error_code,
338  std::size_t /* bytes transferred */)
339 {
340  if (error_code) {
341  if (m_debug_enabled && error_code != boost::asio::error::operation_aborted) {
342  std::cerr << "Receive error: " << error_code << std::endl;
343  }
344  return;
345  }
346 
347  if (m_debug_enabled) {
348  std::cerr << "RX message received." << std::endl;
349  }
350 
351  if (m_handler) {
352  m_message_unpacker.buffer_consumed(m_message_length);
353  msgpack::unpacked result;
354 
355  while (m_message_unpacker.next(&result)) {
356  wamp_message::message_fields fields;
357  result.get().convert(fields);
358 
359  wamp_message message(std::move(fields), std::move(*(result.zone())));
360  if (m_debug_enabled) {
361  std::cerr << "RX message: " << message << std::endl;
362  }
363 
364  m_handler->on_message(std::move(message));
365  }
366  } else {
367  std::cerr << "RX message ignored: no handler attached" << std::endl;
368  }
369 
370  receive_message();
371 }
