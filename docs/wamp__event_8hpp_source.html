101  /*!
102  * Convert and assign the positional arguments to a given list of individual parameters.
103  *
104  * Enables a syntax that lets you declare variables individually, but list them in a
105  * single space to empathize parameter order. This will also throw if the number of
106  * arguments to the invocation doesn't match the number of given parameters.
107  *
108  * Example:
109  * ```
110  * uint64_t id;
111  * std::string name;
112  * invocation->get_each_argument(id, name);
113  * ```
114  *
115  * @throw std::bad_cast
116  */
117  template <typename... T>

120  /*!
121  * The keyword argument published by the event with the given @p key, converted to type T.
122  *
123  * Overloads are provided for `std::string` and `char*` as @p key type.
124  *
125  * This function uses key string comparisons to find the matching value, O(n) with n being
126  * the number of map elements. Memory allocation for keys is avoided though. For larger maps,
127  * you might want to prioritize look-up performance by using `std::map`, `std::unordered_map`
128  * or custom types with custom deserialization. To do this, use kw_arguments<Map>() or
129  * get_kw_arguments<Map>(map), then access the items from there.
130  *
131  * Example:
132  * `std::string id = event.kw_argument<std::string>("id");`
133  *
134  * @throw std::out_of_range
135  * @throw std::bad_cast
136  */
137  template <typename T>
138  T kw_argument(const std::string& key) const;

143  /*!
144  * The keyword argument published by the event with the given @p key, converted to type T,
145  * or the given @p fallback if no such key was passed.
146  *
147  * Overloads are provided for `std::string` and `char*` as @p key type.
148  *
149  * This function uses key string comparisons to find the matching value, O(n) with n being
150  * the number of map elements. Memory allocation for keys is avoided though. For larger maps,
151  * you might want to prioritize look-up performance by using `std::map`, `std::unordered_map`
152  * or custom types with custom deserialization. To do this, use kw_arguments<Map>() or
153  * get_kw_arguments<Map>(map), then access the items from there.
154  *
155  * Example:
156  * `std::string id = event.kw_argument_or("id", std::string());`
157  *
158  * @throw std::bad_cast
159  */
160  template <typename T>
